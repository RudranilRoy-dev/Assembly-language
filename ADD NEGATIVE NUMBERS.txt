LXI H,2050H     ; HL -> count location
MOV C,M         ; C = N
INX H           ; HL -> first element

MVI D,00H       ; D = sum = 0

LOOP: MOV A,M   ; Load element in A
RAL             ; MSB goes to Carry
JNC SKIP        ; If Carry=0, positive so skip

MOV A,D         ; A = sum
ADD M           ; sum = sum + negative number
MOV D,A         ; store sum back in D

SKIP: INX H     ; next element
DCR C           ; count--
JNZ LOOP        ; repeat

MOV A,D         ; final sum
STA 2060H       ; store sum of negatives
HLT
